## Usage

```tsx
function CanvasFractalGridExample() {
  return (
    <div className="grid h-screen w-full pl-[56px] relative">
      <CanvasFractalGrid
        dotSize={3}
        dotSpacing={20}
        dotOpacity={0.4}
        gradientAnimationDuration={6}
        waveIntensity={60}
        waveRadius={250}
        enableGradient={true}
        enableMouseGlow={true}
        enableNoise={true}
      />
    </div>
  )
}

export default CanvasFractalGridExample
```
